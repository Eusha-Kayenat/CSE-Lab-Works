import java.util.Scanner;
public class Task7{
  public static void main (String[]args){
    Scanner sc = new Scanner(System.in);
    System.out.println("Enter BRACU Student ID :");
    int id = sc.nextInt();
    int enrollment_year = id/1000000;
    int session = (id/100000)%10;
    if (session==1){
      System.out.print("Student Joined BRAC in Spring " + enrollment_year);
    }
    else if(session==2){
      System.out.print("Student Joined BRAC in Fall " + enrollment_year);
    }
    else{
      System.out.print("Student Joined BRAC in Summer " + enrollment_year);
    }
  }
}
