import java.util.Scanner;
public class Task11{
  public static void main(String[]args){
  Scanner sc=new Scanner(System.in);
  System.out.println("Enter the value of n: ");
  int n=sc.nextInt();
  int count=1;
  int i=n/10;
  while (i>0){
   i=i/10;
   count++;     
  }
  System.out.print("Total digits = "+count);
  }
 }
