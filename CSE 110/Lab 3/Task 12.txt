import java.util.Scanner;
public class Task12{
  public static void main(String[]args){
  Scanner sc=new Scanner(System.in);
  System.out.println("Enter the value of n: ");
  int n=sc.nextInt();
   while (n>0){
     int digit =n%10;
     if (n<10){
      System.out.print(n);
     }
     else{
      System.out.print(digit+",");
     }
     n=n/10;
   } 
  }
}
